```java
//기본 연산자 
public class Main {
    public static void main(String[] args) {
        int a,b,c;
        a = 3;
        b = 4;
        c = a+b;

        System.out.println(a+"+"+b+"="+c);
        c = a-b;
        System.out.println(a+"-"+b+"="+c);
        c = a*b;
        System.out.println(a+"*"+b+"="+c);
        c = a/b;
        System.out.println(a+"/"+b+"="+c);
        c = a%b;
        System.out.println(a+"%"+b+"="+c);
    }
}
```
