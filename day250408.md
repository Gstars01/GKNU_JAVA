```java
package circle;

public class circle {
    double radius;
    String name;

    // 생성자
    public circle() {
        
    }


    double getArea() {
        return 3.14 * radius * radius;
    }

    public static void main(String[] args) {
    	//객체 생성 
        circle pizza = new circle();
        
        // 객체의 값 설정
        pizza.radius = 4.0;
        pizza.name = "도미노 피자";


        System.out.println("피자 이름: " + pizza.name + ", 반지름: " + pizza.radius);
        System.out.println("피자의 면적: " + pizza.getArea());
    }
}
```
